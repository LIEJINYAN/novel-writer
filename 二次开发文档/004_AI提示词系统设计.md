# AI 提示词系统设计

## 1. 系统概述

PySide6 版的 AI 提示词系统是对原版 CLI + AI 助手命令模板的全面升级。原版依赖外部 AI 助手（Claude Code、Cursor 等）的斜杠命令系统，PySide6 版则**直接调用 AI API**，拥有完整的提示词管理能力。

### 1.1 与原版的核心差异

| 维度 | 原版（CLI + AI 助手） | PySide6 版（直接 API 调用） |
|------|---------------------|---------------------------|
| **AI 调用方式** | 通过 AI 助手的斜杠命令 | 直接调用 AI API |
| **提示词管理** | Markdown/TOML 静态模板 | Python 动态模板 + 数据库存储 |
| **上下文管理** | 文件系统读取 | 数据库 + 文件混合 |
| **温度控制** | 不可配置（低温锁定） | 完全可配置 |
| **模型选择** | 由 AI 助手决定 | 用户自由切换 |
| **流式输出** | 不支持 | 支持流式输出 |
| **多模型** | 不支持 | 支持多模型切换/对比 |
| **本地模型** | 不支持 | 支持 Ollama 本地模型 |

### 1.2 AI 架构总览

## 2. AI 提供商设计

### 2.1 提供商基类

```python
# core/ai/providers/base.py
from abc import ABC, abstractmethod
from typing import AsyncIterator, List, Optional
from dataclasses import dataclass

@dataclass
class Message:
    role: str  # system / user / assistant
    content: str

@dataclass
class AIResponse:
    content: str
    model: str
    usage: dict  # {prompt_tokens, completion_tokens, total_tokens}
    finish_reason: str

@dataclass
class AIConfig:
    temperature: float = 0.7
    max_tokens: int = 4000
    top_p: float = 1.0
    frequency_penalty: float = 0.0
    presence_penalty: float = 0.0
    stop: Optional[List[str]] = None

class AIProvider(ABC):
    """AI 提供商抽象基类"""
    
    name: str           # 提供商标识（如 "openai"）
    display_name: str   # 显示名称（如 "OpenAI"）
    
    @abstractmethod
    async def chat(
        self,
        messages: List[Message],
        config: AIConfig = None,
        model: str = None
    ) -> AIResponse:
        """同步对话"""
        ...
    
    @abstractmethod
    async def chat_stream(
        self,
        messages: List[Message],
        config: AIConfig = None,
        model: str = None
    ) -> AsyncIterator[str]:
        """流式对话，逐字返回"""
        ...
    
    @abstractmethod
    def get_available_models(self) -> List[str]:
        """获取可用模型列表"""
        ...
    
    @abstractmethod
    def validate_config(self) -> bool:
        """验证配置是否有效（API Key 等）"""
        ...
```

### 2.2 OpenAI 提供商

```python
# core/ai/providers/openai.py
import openai
from .base import AIProvider, Message, AIConfig, AIResponse

class OpenAIProvider(AIProvider):
    name = "openai"
    display_name = "OpenAI"
    
    MODELS = [
        "gpt-4o", "gpt-4o-mini",
        "gpt-4-turbo", "gpt-4",
        "gpt-3.5-turbo",
    ]
    
    def __init__(self, api_key: str, api_base: str = None):
        self.client = openai.AsyncOpenAI(
            api_key=api_key,
            base_url=api_base
        )
    
    async def chat(self, messages, config=None, model=None):
        model = model or "gpt-4o"
        config = config or AIConfig()
        
        response = await self.client.chat.completions.create(
            model=model,
            messages=[{"role": m.role, "content": m.content} for m in messages],
            temperature=config.temperature,
            max_tokens=config.max_tokens,
            top_p=config.top_p,
            frequency_penalty=config.frequency_penalty,
            presence_penalty=config.presence_penalty,
            stop=config.stop,
        )
        
        return AIResponse(
            content=response.choices[0].message.content,
            model=response.model,
            usage={
                "prompt_tokens": response.usage.prompt_tokens,
                "completion_tokens": response.usage.completion_tokens,
                "total_tokens": response.usage.total_tokens,
            },
            finish_reason=response.choices[0].finish_reason,
        )
    
    async def chat_stream(self, messages, config=None, model=None):
        model = model or "gpt-4o"
        config = config or AIConfig()
        
        stream = await self.client.chat.completions.create(
            model=model,
            messages=[{"role": m.role, "content": m.content} for m in messages],
            temperature=config.temperature,
            max_tokens=config.max_tokens,
            stream=True,
        )
        
        async for chunk in stream:
            if chunk.choices[0].delta.content:
                yield chunk.choices[0].delta.content
    
    def get_available_models(self):
        return self.MODELS
    
    def validate_config(self):
        return bool(self.client.api_key)
```

### 2.3 Anthropic Claude 提供商

```python
# core/ai/providers/anthropic.py
import anthropic
from .base import AIProvider, Message, AIConfig, AIResponse

class AnthropicProvider(AIProvider):
    name = "anthropic"
    display_name = "Anthropic Claude"
    
    MODELS = [
        "claude-sonnet-4-5-20250929",
        "claude-opus-4-20250514",
        "claude-3-5-haiku-20241022",
    ]
    
    def __init__(self, api_key: str):
        self.client = anthropic.AsyncAnthropic(api_key=api_key)
    
    async def chat(self, messages, config=None, model=None):
        model = model or "claude-sonnet-4-5-20250929"
        config = config or AIConfig()
        
        # 分离 system 消息
        system_msg = ""
        chat_messages = []
        for m in messages:
            if m.role == "system":
                system_msg += m.content + "\n"
            else:
                chat_messages.append({"role": m.role, "content": m.content})
        
        response = await self.client.messages.create(
            model=model,
            system=system_msg.strip(),
            messages=chat_messages,
            temperature=config.temperature,
            max_tokens=config.max_tokens,
            top_p=config.top_p,
            stop_sequences=config.stop,
        )
        
        return AIResponse(
            content=response.content[0].text,
            model=response.model,
            usage={
                "prompt_tokens": response.usage.input_tokens,
                "completion_tokens": response.usage.output_tokens,
                "total_tokens": response.usage.input_tokens + response.usage.output_tokens,
            },
            finish_reason=response.stop_reason,
        )
    
    async def chat_stream(self, messages, config=None, model=None):
        model = model or "claude-sonnet-4-5-20250929"
        config = config or AIConfig()
        
        system_msg = ""
        chat_messages = []
        for m in messages:
            if m.role == "system":
                system_msg += m.content + "\n"
            else:
                chat_messages.append({"role": m.role, "content": m.content})
        
        async with self.client.messages.stream(
            model=model,
            system=system_msg.strip(),
            messages=chat_messages,
            temperature=config.temperature,
            max_tokens=config.max_tokens,
        ) as stream:
            async for text in stream.text_stream:
                yield text
    
    def get_available_models(self):
        return self.MODELS
    
    def validate_config(self):
        return bool(self.client.api_key)
```

### 2.4 Ollama 本地模型提供商

```python
# core/ai/providers/ollama.py
import aiohttp
from .base import AIProvider, Message, AIConfig, AIResponse

class OllamaProvider(AIProvider):
    name = "ollama"
    display_name = "Ollama (本地模型)"
    
    def __init__(self, api_base: str = "http://localhost:11434"):
        self.api_base = api_base.rstrip("/")
    
    async def chat(self, messages, config=None, model=None):
        model = model or "qwen2.5:14b"
        config = config or AIConfig()
        
        async with aiohttp.ClientSession() as session:
            response = await session.post(
                f"{self.api_base}/api/chat",
                json={
                    "model": model,
                    "messages": [{"role": m.role, "content": m.content} for m in messages],
                    "options": {
                        "temperature": config.temperature,
                        "num_predict": config.max_tokens,
                        "top_p": config.top_p,
                    },
                    "stream": False,
                }
            )
            data = await response.json()
            
            return AIResponse(
                content=data["message"]["content"],
                model=data["model"],
                usage={
                    "prompt_tokens": data.get("prompt_eval_count", 0),
                    "completion_tokens": data.get("eval_count", 0),
                    "total_tokens": data.get("prompt_eval_count", 0) + data.get("eval_count", 0),
                },
                finish_reason="stop",
            )
    
    async def chat_stream(self, messages, config=None, model=None):
        model = model or "qwen2.5:14b"
        config = config or AIConfig()
        
        async with aiohttp.ClientSession() as session:
            response = await session.post(
                f"{self.api_base}/api/chat",
                json={
                    "model": model,
                    "messages": [{"role": m.role, "content": m.content} for m in messages],
                    "options": {
                        "temperature": config.temperature,
                        "num_predict": config.max_tokens,
                        "top_p": config.top_p,
                    },
                    "stream": True,
                }
            )
            
            async for line in response.content:
                import json
                data = json.loads(line)
                if data.get("message", {}).get("content"):
                    yield data["message"]["content"]
                if data.get("done"):
                    break
    
    async def get_available_models(self):
        async with aiohttp.ClientSession() as session:
            response = await session.get(f"{self.api_base}/api/tags")
            data = await response.json()
            return [m["name"] for m in data.get("models", [])]
    
    def validate_config(self):
        # Ollama 不需要 API Key，只需检查服务是否运行
        return True
```

### 2.5 AI 管理器

```python
# core/ai/ai_manager.py
from typing import Dict, Optional
from .providers.base import AIProvider, AIConfig, Message

class AIManager:
    """AI 提供商管理器（单例）"""
    
    _instance = None
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
        return cls._instance
    
    def __init__(self):
        self._providers: Dict[str, AIProvider] = {}
        self._current_provider: str = "openai"
        self._current_model: str = ""
        self._default_config: AIConfig = AIConfig()
    
    def register_provider(self, provider: AIProvider):
        """注册 AI 提供商"""
        self._providers[provider.name] = provider
    
    def get_provider(self, name: str = None) -> AIProvider:
        """获取提供商"""
        name = name or self._current_provider
        return self._providers.get(name)
    
    def set_current_provider(self, name: str):
        """切换当前提供商"""
        if name in self._providers:
            self._current_provider = name
    
    def set_current_model(self, model: str):
        """设置当前模型"""
        self._current_model = model
    
    def set_config(self, config: AIConfig):
        """设置默认参数"""
        self._default_config = config
    
    async def generate(
        self,
        prompt: str,
        system_prompt: str = "",
        config: AIConfig = None,
        provider: str = None,
        model: str = None
    ) -> str:
        """生成文本（同步）"""
        provider_obj = self.get_provider(provider)
        messages = []
        if system_prompt:
            messages.append(Message(role="system", content=system_prompt))
        messages.append(Message(role="user", content=prompt))
        
        response = await provider_obj.chat(
            messages=messages,
            config=config or self._default_config,
            model=model or self._current_model
        )
        return response.content
    
    async def generate_stream(
        self,
        prompt: str,
        system_prompt: str = "",
        config: AIConfig = None,
        provider: str = None,
        model: str = None
    ):
        """生成文本（流式）"""
        provider_obj = self.get_provider(provider)
        messages = []
        if system_prompt:
            messages.append(Message(role="system", content=system_prompt))
        messages.append(Message(role="user", content=prompt))
        
        async for chunk in provider_obj.chat_stream(
            messages=messages,
            config=config or self._default_config,
            model=model or self._current_model
        ):
            yield chunk
    
    def list_providers(self) -> list:
        """列出所有已注册的提供商"""
        return [
            {
                "name": p.name,
                "display_name": p.display_name,
                "models": p.get_available_models(),
                "is_valid": p.validate_config(),
            }
            for p in self._providers.values()
        ]
```

## 3. 提示词模板系统

### 3.1 模板分类

PySide6 版将原版的 50+ 个 Markdown/TOML 模板转换为 Python 动态模板，分为以下类别：

| 类别 | 模板数量 | 说明 | 对应原版 |
|------|---------|------|---------|
| **写作模板** | 7 | 七步方法论命令 | `/constitution` ~ `/analyze` |
| **编辑模板** | 5 | 续写、润色、改写、对话生成 | 新增 |
| **分析模板** | 7 | 情节检查、时间线、关系追踪等 | `/track` ~ `/checklist` |
| **专家模板** | 4 | 剧情、角色、世界观、风格专家 | `/expert` 系列 |
| **插件模板** | 25+ | 插件扩展命令 | 插件命令 |

### 3.2 模板基类

```python
# core/ai/prompt_templates/base.py
from dataclasses import dataclass, field
from typing import Dict, Any
import re

@dataclass
class PromptTemplate:
    """提示词模板基类"""
    
    name: str                           # 模板名称
    description: str                    # 模板描述
    system_prompt: str = ""             # 系统提示词
    user_prompt_template: str = ""      # 用户提示词模板（支持变量）
    default_config: AIConfig = field(default_factory=AIConfig)
    
    def render(self, **kwargs) -> str:
        """渲染模板，替换变量"""
        result = self.user_prompt_template
        # 简单变量替换 {variable_name}
        for key, value in kwargs.items():
            result = result.replace(f"{{{key}}}", str(value))
        return result
    
    def build_messages(self, **kwargs) -> List[Message]:
        """构建完整的消息列表"""
        messages = []
        if self.system_prompt:
            messages.append(Message(role="system", content=self.system_prompt))
        messages.append(Message(role="user", content=self.render(**kwargs)))
        return messages
```

```python
# utils/token_counter.py
import tiktoken

class TokenCounter:
    """Token 计数器"""
    
    def __init__(self, model: str = "gpt-4o"):
        try:
            self.encoder = tiktoken.encoding_for_model(model)
        except KeyError:
            self.encoder = tiktoken.get_encoding("cl100k_base")
    
    def count(self, text: str) -> int:
        """计算文本的 token 数"""
        return len(self.encoder.encode(text))
    
    def count_messages(self, messages: List[Message]) -> int:
        """计算消息列表的总 token 数"""
        total = 0
        for msg in messages:
            total += 4  # 每条消息固定开销
            total += self.count(msg.content)
        total += 2  # 对话结束标记
        return total
    
    def truncate_to_tokens(self, text: str, max_tokens: int) -> str:
        """截断文本到指定 token 数"""
        tokens = self.encoder.encode(text)
        if len(tokens) <= max_tokens:
            return text
        return self.encoder.decode(tokens[:max_tokens])
```

## 7. 错误处理与重试

### 7.1 错误处理框架

```python
# core/ai/error_handler.py
import asyncio
from functools import wraps

class AIError(Exception):
    """AI 调用基础异常"""
    pass

class RateLimitError(AIError):
    """速率限制异常"""
    def __init__(self, retry_after: int = 60):
        self.retry_after = retry_after
        super().__init__(f"请求过于频繁，请 {retry_after} 秒后重试")

class AuthError(AIError):
    """认证失败异常"""
    pass

class ModelNotAvailableError(AIError):
    """模型不可用异常"""
    pass

class ContextTooLongError(AIError):
    """上下文过长异常"""
    pass


def with_retry(
    max_retries: int = 3,
    base_delay: float = 1.0,
    max_delay: float = 60.0
):
    """指数退避重试装饰器"""
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            last_error = None
            for attempt in range(max_retries):
                try:
                    return await func(*args, **kwargs)
                except RateLimitError as e:
                    last_error = e
                    delay = min(e.retry_after, max_delay)
                    await asyncio.sleep(delay)
                except (ConnectionError, TimeoutError) as e:
                    last_error = e
                    delay = min(base_delay * (2 ** attempt), max_delay)
                    await asyncio.sleep(delay)
                except AuthError:
                    raise  # 认证错误不重试
                except Exception as e:
                    last_error = e
                    if attempt < max_retries - 1:
                        delay = base_delay * (2 ** attempt)
                        await asyncio.sleep(delay)
                    else:
                        raise
            raise last_error
        return wrapper
    return decorator
```

### 7.2 错误码处理

| 状态码 | 异常类型 | 处理方式 |
|--------|---------|---------|
| 400 | `ValueError` | 检查请求参数 |
| 401 | `AuthError` | 提示用户检查 API Key |
| 404 | `ModelNotAvailableError` | 提示模型不可用 |
| 429 | `RateLimitError` | 指数退避重试 |
| 500 | `AIError` | 重试，超过3次提示用户 |
| 超时 | `TimeoutError` | 重试 |
| 网络断开 | `ConnectionError` | 提示检查网络 |

### 8.2 AI 内容审计

```python
# core/ai/services/audit_service.py

class AIAuditService:
    """AI 内容审计服务"""
    
    BANNED_WORDS = [
        "弥漫着", "唯一的", "直到", "不禁", "顿时",
        "摇摇欲坠", "空气凝固", "话音未落", "猛地",
        "宛如", "仿佛", "犹如", "心中暗想",
    ]
    
    def audit_text(self, text: str) -> dict:
        """审计文本的AI痕迹"""
        issues = []
        
        # 1. 检查AI高频词
        for word in self.BANNED_WORDS:
            if word in text:
                issues.append({
                    "type": "banned_word",
                    "word": word,
                    "severity": "high",
                    "suggestion": "替换为更具体的表达",
                })
        
        # 2. 检查段落长度
        paragraphs = text.split("\n\n")
        for i, para in enumerate(paragraphs):
            char_count = len(para.strip())
            if char_count > 150:
                issues.append({
                    "type": "long_paragraph",
                    "paragraph": i + 1,
                    "char_count": char_count,
                    "severity": "medium",
                    "suggestion": f"段落 {i+1} 过长（{char_count}字），建议拆分为50-100字短段",
                })
        
        # 3. 检查单句成段比例
        single_sentence_paras = sum(1 for p in paragraphs if p.count("。") <= 1)
        ratio = single_sentence_paras / len(paragraphs) if paragraphs else 0
        if ratio < 0.3:
            issues.append({
                "type": "low_single_sentence_ratio",
                "ratio": f"{ratio:.0%}",
                "severity": "medium",
                "suggestion": f"单句成段比例仅 {ratio:.0%}，建议提升到 30-50%",
            })
        
        return {
            "total_issues": len(issues),
            "issues": issues,
            "ai_score": self._calculate_ai_score(issues, len(text)),
        }
```
