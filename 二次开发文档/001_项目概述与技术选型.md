# 项目概述与技术选型

## 1. 项目简介

### 1.1 项目名称
**Novel Writer PySide6 版** - AI 原生小说创作桌面系统

### 1.2 一句话简介
基于 PySide6 的 AI 原生中文小说创作桌面软件 —— 集成多种写作方法论、智能 AI 辅助、完整追踪系统、插件化扩展的沉浸式写作平台。

### 1.3 项目定位
- **原版定位**：CLI 工具 + AI 助手命令模板（面向技术型写作者）
- **PySide6 版定位**：独立桌面应用 + 原生 AI 集成（面向所有写作者）

### 1.4 核心价值
1. **沉浸式写作体验**：原生桌面应用，无浏览器干扰
2. **AI 原生设计**：深度集成 AI 能力，智能辅助创作
3. **方法论驱动**：内置多种经典写作方法论，科学创作
4. **完整追踪系统**：情节、角色、关系、时间线全方位追踪
5. **本地优先**：数据本地存储，隐私安全，离线可用
6. **插件化扩展**：灵活的插件系统，功能可定制

## 2. 技术选型

### 2.1 为什么选择 PySide6？

| 选型维度 | PySide6 优势 | 原版 (TypeScript/Node CLI) |
|---------|-------------|--------------------------|
| **AI 生态** | Python 有最强的 AI/ML 生态（LangChain、Transformers、Ollama 等） | Node.js AI 生态较弱 |
| **原生体验** | 真正的原生桌面应用，性能好，启动快 | CLI 工具，无 GUI |
| **系统集成** | 托盘、全局快捷键、文件关联、拖拽等完整系统集成 | 仅限终端 |
| **离线能力** | 可集成本地大模型，完全离线使用 | 需要 AI 助手工具 |
| **用户群体** | 面向所有写作者，门槛低 | 面向技术型写作者 |
| **富文本编辑** | QTextEdit 原生支持，可定制性强 | 无编辑器 |
| **数据存储** | SQLite 本地数据库，查询高效 | JSON/YAML 文件 |

#### EPUB 导出实现

```python
# services/exporters/epub_exporter.py
from ebooklib import epub
from typing import List
from models import Chapter, Project

class EpubExporter:
    """EPUB 电子书导出器。"""

    def export(self, project: Project, chapters: List[Chapter],
               output_path: str, options: dict = None) -> str:
        """导出为 EPUB。"""
        book = epub.EpubBook()

        # 元数据
        book.set_identifier(f"novel-{project.id}")
        book.set_title(project.name)
        book.set_language("zh-CN")
        if project.description:
            book.add_author(project.description[:100])

        # CSS 样式
        css = """
        @namespace epub "http://www.idpf.org/2007/ops";
        body { font-family: "Noto Serif CJK SC", serif; line-height: 1.8; }
        h1 { text-align: center; margin: 2em 0; }
        p { text-indent: 2em; margin: 0.5em 0; }
        """
        style = epub.EpubItem(
            uid="style", file_name="style.css",
            media_type="text/css", content=css,
        )
        book.add_item(style)

        # 章节
        for i, chapter in enumerate(chapters):
            epub_chapter = epub.EpubHtml(
                title=chapter.title,
                file_name=f"chapter_{i+1:03d}.xhtml",
                lang="zh-CN",
            )
            epub_chapter.content = f"""
            <html><head><link rel="stylesheet" type="text/css" href="style.css"/>
            </head><body>
            <h1>{chapter.title}</h1>
            {chapter.content}
            </body></html>
            """
            book.add_item(epub_chapter)
            book.toc.append(epub.Link(
                f"chapter_{i+1:03d}.xhtml",
                chapter.title, f"ch{i+1}",
            ))
            book.add_item(epub_chapter)
            book.spine.append(epub_chapter)

        # 目录
        book.add_item(epub.EpubNcx())
        book.add_item(epub.EpubNav())

        # 输出
        epub.write_epub(output_path, book)
        return output_path
```

```python
# services/exporters/pdf_exporter.py
from fpdf import FPDF
from typing import List

class PDFExporter:
    """PDF 文档导出器（使用 fpdf2）。"""

    def export(self, title: str, chapters: List[dict],
               output_path: str, options: dict = None) -> str:
        """导出为 PDF。"""
        font_path = options.get("font_path", "")
        paper_size = options.get("paper_size", "A4")

        pdf = FPDF(orientation="P", unit="mm", format=paper_size)
        pdf.add_font("CJK", "", font_path or "", uni=True)
        pdf.set_auto_page_break(auto=True, margin=20)

        # 封面
        pdf.add_page()
        pdf.set_font("CJK", "", 24)
        pdf.cell(0, 60, "", new_x="LMARGIN", new_y="NEXT")
        pdf.cell(0, 20, title, align="C", new_x="LMARGIN", new_y="NEXT")

        # 正文
        for chapter in chapters:
            pdf.add_page()
            pdf.set_font("CJK", "", 18)
            pdf.cell(0, 15, chapter["title"], align="C",
                     new_x="LMARGIN", new_y="NEXT")

            pdf.set_font("CJK", "", 12)
            for paragraph in chapter["paragraphs"]:
                # 处理缩进和分段
                pdf.multi_cell(0, 8, f"  {paragraph}")
                pdf.ln(2)

        pdf.output(output_path)
        return output_path
```

#### DOCX 导出实现

```python
# services/exporters/docx_exporter.py
from docx import Document
from docx.shared import Pt, Cm
from docx.enum.text import WD_ALIGN_PARAGRAPH

class DocxExporter:
    """Word 文档导出器。"""

    def export(self, title: str, chapters: List[dict],
               output_path: str, options: dict = None) -> str:
        """导出为 DOCX。"""
        doc = Document()

        # 封面
        title_para = doc.add_paragraph()
        title_para.alignment = WD_ALIGN_PARAGRAPH.CENTER
        run = title_para.add_run(title)
        run.font.size = Pt(22)
        run.bold = True

        doc.add_page_break()

        # 样式设置
        style = doc.styles["Normal"]
        font = style.font
        font.name = "宋体"
        font.size = Pt(12)
        paragraph_format = style.paragraph_format
        paragraph_format.line_spacing = 1.8
        paragraph_format.first_line_indent = Cm(0.75)

        # 章节
        for chapter in chapters:
            heading = doc.add_heading(chapter["title"], level=1)

            for paragraph_text in chapter["paragraphs"]:
                doc.add_paragraph(paragraph_text)

        doc.save(output_path)
        return output_path
```

```python
# ui/dialogs/export_dialog.py
from PySide6.QtWidgets import (
    QDialog, QVBoxLayout, QHBoxLayout, QComboBox,
    QCheckBox, QLabel, QPushButton, QFileDialog,
    QDialogButtonBox, QGroupBox,
)

class ExportDialog(QDialog):
    """导出对话框。"""

    FORMATS = {
        "txt": {"label": "纯文本 (.txt)", "icon": "txt"},
        "md": {"label": "Markdown (.md)", "icon": "md"},
        "html": {"label": "HTML (.html)", "icon": "html"},
        "epub": {"label": "EPUB 电子书 (.epub)", "icon": "epub"},
        "pdf": {"label": "PDF 文档 (.pdf)", "icon": "pdf"},
        "docx": {"label": "Word 文档 (.docx)", "icon": "docx"},
    }

    def __init__(self, project_name: str, parent=None):
        super().__init__(parent)
        self.setWindowTitle("导出项目")
        self.setMinimumWidth(450)
        self._project_name = project_name
        self._init_ui()

    def _init_ui(self) -> None:
        layout = QVBoxLayout(self)

        # 格式选择
        format_group = QGroupBox("导出格式")
        format_layout = QVBoxLayout(format_group)

        self._format_combo = QComboBox()
        for fmt_id, fmt_info in self.FORMATS.items():
            self._format_combo.addItem(fmt_info["label"], fmt_id)
        format_layout.addWidget(self._format_combo)

        # 导出选项
        options_group = QGroupBox("导出选项")
        options_layout = QVBoxLayout(options_group)

        self._include_cover = QCheckBox("包含封面")
        self._include_cover.setChecked(True)
        options_layout.addWidget(self._include_cover)

        self._include_toc = QCheckBox("包含目录")
        self._include_toc.setChecked(True)
        options_layout.addWidget(self._include_toc)

        self._chapters_only = QCheckBox("仅导出选中章节")
        options_layout.addWidget(self._chapters_only)

        layout.addWidget(format_group)
        layout.addWidget(options_group)

        # 按钮
        buttons = QDialogButtonBox()
        self._export_btn = buttons.addButton("导出", QDialogButtonBox.AcceptRole)
        self._cancel_btn = buttons.addButton("取消", QDialogButtonBox.RejectRole)
        buttons.accepted.connect(self._on_export)
        buttons.rejected.connect(self.reject)
        layout.addWidget(buttons)

    def _on_export(self) -> None:
        """选择导出路径并执行导出。"""
        fmt = self._format_combo.currentData()
        ext = fmt

        default_name = f"{self._project_name}.{ext}"
        file_path, _ = QFileDialog.getSaveFileName(
            self, "导出文件", default_name,
            f"{self._format_combo.currentText()} (*.{ext})",
        )
        if file_path:
            self._output_path = file_path
            self.accept()

    def get_export_config(self) -> dict:
        """获取导出配置。"""
        return {
            "format": self._format_combo.currentData(),
            "output_path": getattr(self, "_output_path", ""),
            "include_cover": self._include_cover.isChecked(),
            "include_toc": self._include_toc.isChecked(),
            "chapters_only": self._chapters_only.isChecked(),
        }
```

## 7. 技术风险与应对

### 7.1 风险识别

| 风险 | 影响 | 概率 | 应对措施 |
|------|------|------|---------|
| PySide6 学习曲线 | 开发进度 | 中 | 先做最小可用版本，逐步优化 |
| 富文本编辑器复杂度 | 功能完整性 | 高 | 先实现基础功能，后续迭代增强 |
| AI  API 成本 | 使用成本 | 中 | 支持本地模型，提供多种价位选择 |
| 数据迁移 | 用户采纳 | 低 | 提供原版数据导入工具 |
| 打包体积 | 分发成本 | 低 | UPX 压缩，按需打包 |

### 7.2 关键技术决策

1. **QWidget vs QML**：选择 QWidget，更适合传统桌面应用，性能更好
2. **SQLAlchemy vs 原生 SQL**：选择 SQLAlchemy，ORM 更易维护
3. **LangChain vs 直接调用**：初期直接调用，后期引入 LangChain
4. **打包工具**：PyInstaller，生态成熟，文档丰富
